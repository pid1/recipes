  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

% Redefine \section to include numbering in TOC but hide it on the page
\let\oldsection\section
\renewcommand{\section}[1]{%
    \newpage
    {\centering\Huge\bfseries #1\par} % Display section title without number
    \addcontentsline{toc}{section}{\thesection\quad #1} % Add numbered entry to TOC
    \stepcounter{section} % Increment section counter
}

% Redefine \subsection to include numbering in TOC but hide it on the page
\let\oldsubsection\subsection
\renewcommand{\subsection}[1]{%
    \newpage
    {\centering\large\bfseries #1\par} % Display subsection title without number
    \addcontentsline{toc}{subsection}{\thesubsection\quad #1} % Add numbered entry to TOC
    \stepcounter{subsection} % Increment subsection counter
}
